\begin{equation}
{}^0\boldsymbol{R}_2={}^0\boldsymbol{R}_1{}^1\boldsymbol{R}_2.

\begin{align*}

	\cos(\alpha\pm\beta)=&\cos\alpha\cos\beta\pm\sin\alpha\sin\beta\\
	\tan(\alpha\pm\beta)=&\frac{\tan\alpha\pm\tan\beta}{1\pm\tan\alpha\tan\beta}.
\end{align*}
